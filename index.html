/**
* @description Abstract class which can be adopted by subclass to support Cache.OrgPartition management
* <p> Coverage: DynamicScriptingServiceTest, DynamicScriptingTriggerHelperTest </p>
* @author Radoslaw Wilczynski
**/
public abstract class BaseCache {

 	/**
 	* @description Returns prepared map of objects that should be put in Cache.OrgPartition. Map's key is a string representing a name/key under which
 	* object will be stored in Cache.OrgPartition. Value is object which will be put in cache under a key.
 	* @author Radoslaw Wilczynski
 	* @param keys Set of keys for which items to put to Cache.OrgPartition will be prepared
 	* @return Map of object to put into Cache.OrgPartition
 	**/
 	protected abstract Map<String, Object> prepareItemsToPut(Set<String> keys);


 	/**
 	* @description Return objects from Cache.OrgPartition mapped by unique key. Any key not found in cache will be included in call to
 	* putToCache method where object to cache will be prepared.
 	* @author Radoslaw Wilczynski
 	* @param keys Set of keys for which items to put to Cache.OrgPartition will be prepared
 	* @return Map of objects stored in Cache.OrgPartition under unique keys
 	**/
 	protected Map<String, Object> get(Set<String> keys) {

 	}
}

NOTES

  • List of test classes which are providing code coverage should be provided in @description part of class like in the example above, surrounded by <p> tag
  • If you have more than one paragraph in the doc comment, separate the paragraphs with a paragraph tag, as shown.
  • Only elements with global, public and protected accessors will be generated.
  • At least description should be written for public inner classes.
